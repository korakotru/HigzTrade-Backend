dotnet add package Serilog.AspNetCore  // ตัวหลักสำหรับ ASP.NET Core (รวม sinks console/file)
dotnet add package Serilog.Sinks.Seq  // ถ้าอยาก log ไป Seq (แนะนำมาก)
dotnet add package Serilog.Sinks.File  // log ไฟล์ JSON
dotnet add package Serilog.Settings.Configuration  // อ่าน config จาก appsettings.json
dotnet add package Serilog.Enrichers.Environment  // เพิ่ม MachineName, ThreadId
dotnet add package Serilog.Formatting.Compact // เพื่อให้ log เป็น format .clef ซึ่งเป็น JSON พิเศษที่ต้องมีฟิลด์บังคับอย่าง @t (เวลา) และ @mt (ข้อความ)เพื่อให้มันแสดงผลได้สวยงามเหมือนใน Seq (ใช้โปรแกรม Compact Log Viewer ดูได้)
